<button id="hideBtn">Hide</button>
          <button id="showBtn">Show</button>
          <div id="box" style="width: 100px; height: 100px; background-color: blue"></div>
  
          <script>
            $("#hideBtn").click(function () {
              $("#box").hide(); // Hides the element instantly
            });
            $("#showBtn").click(function () {
              $("#box").show(); // Shows the element instantly
            });
          </script>
        
    
        
          <button id="fadeInBtn">Fade In</button>
          <button id="fadeOutBtn">Fade Out</button>
          <div id="box2" style="width: 100px; height: 100px; background-color: red; display: none"></div>
  
          <script>
            $("#fadeInBtn").click(function () {
              $("#box2").fadeIn(1000); // Fades in over 1 second
            });
            $("#fadeOutBtn").click(function () {
              $("#box2").fadeOut(1000); // Fades out over 1 second
            });
          </script>
        
    
        
          <button id="toggleBtn">Toggle</button>
          <div id="box3" style="width: 100px; height: 100px; background-color: green"></div>
  
          <script>
            $("#toggleBtn").click(function () {
              $("#box3").toggle(); // Toggles the visibility
            });
          </script>
        
    
        
          <button id="slideUpBtn">Slide Up</button>
          <button id="slideDownBtn">Slide Down</button>
          <div id="box4" style="width: 100px; height: 100px; background-color: purple"></div>
  
          <script>
            $("#slideUpBtn").click(function () {
              $("#box4").slideUp(); // Slides up
            });
            $("#slideDownBtn").click(function () {
              $("#box4").slideDown(); // Slides down
            });
          </script>
        
    
        
          <button id="fadeOutWithCallback">Fade Out with Callback</button>
          <div id="box5" style="width: 100px; height: 100px; background-color: orange"></div>
  
          <script>
            $("#fadeOutWithCallback").click(function () {
              $("#box5").fadeOut(1000, function () {
                alert("Fade out complete!");
              });
            });
          </script>
        
    
        
          <button id="fadeToggleBtn">Fade Toggle (5 seconds)</button>
          <button id="slideToggleBtn">Slide Toggle (5 seconds)</button>
          <div id="box6" style="width: 100px; height: 100px; background-color: cyan"></div>
  
          <script>
            $("#fadeToggleBtn").click(function () {
              $("#box6").fadeToggle(5000); // Fades in/out
            });
            $("#slideToggleBtn").click(function () {
              $("#box6").slideToggle(5000); // Slides up/down
            });
          </script>
        
    
        
          <button id="animateBtn">Animate</button>
          <div id="box7" style="width: 100px; height: 100px; background-color: pink; position: relative"></div>
  
          <script>
            $("#animateBtn").click(function () {
              $("#box7").animate(
                {
                  width: "200px",
                  height: "200px",
                  left: "100px",
                },
                1000
              ); // Animates over 1 second
            });
          </script>
