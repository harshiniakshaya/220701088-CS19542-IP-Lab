<button id="clickBtn">Click Me</button>
  
          <script>
            $("#clickBtn").on("click", function () {
              alert("Button clicked!");
            });
          </script>
        
    
        
          <div id="hoverBox" style="width: 100px; height: 100px; background-color: blue"></div>
  
          <script>
            $("#hoverBox").on("mouseenter", function () {
              $(this).css("background-color", "green"); // Changes to green on hover
            });
  
            $("#hoverBox").on("mouseleave", function () {
              $(this).css("background-color", "blue"); // Changes back to blue when hover ends
            });
          </script>
        
    
        
          <input type="text" id="inputField" placeholder="Type something..." />
  
          <script>
            $("#inputField").on("keypress", function (event) {
              console.log("Key pressed: " + event.key);
            });
          </script>
        
    
        
          <input type="text" id="focusField" placeholder="Click to focus" />
  
          <script>
            $("#focusField").on("focus", function () {
              $(this).css("background-color", "lightblue"); // Changes color on focus
            });
  
            $("#focusField").on("blur", function () {
              $(this).css("background-color", ""); // Resets color on blur
            });
          </script>

jQuery Chaining

Chaining allows you to apply multiple jQuery methods to the same element(s) in a single statement, making code cleaner and more efficient. 

        
          <div id="box" style="width: 100px; height: 100px; background-color: red"></div>
          <button id="chainBtn">Apply Chaining</button>
  
          <script>
            $("#chainBtn").on("click", function () {
              $("#box")
                .css("background-color", "yellow") // Changes color to yellow
                .slideUp(500) // Slides up in 500ms
                .slideDown(500) // Slides back down in 500ms
                .fadeOut(500) // Fades out in 500ms
                .fadeIn(500); // Fades back in in 500ms
            });
          </script>
